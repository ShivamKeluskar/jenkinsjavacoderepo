node('Slavenode2') {  // Make sure the node label is correct
    try {
        stage('Checkout') {
            git branch: 'main', url: 'https://github.com/ShivamKeluskar/jenkinsjavacoderepo.git'
        }

        stage('Build') {
            dir('Javarepo1') {  // Make sure this matches your project folder structure
                echo 'Building the project...'
                bat 'mvn clean package'
            }
        }

        stage('Archive Artifacts') {
            archiveArtifacts artifacts: '**/target/*.jar', onlyIfSuccessful: true
        }

        echo 'Build succeeded!'

    } catch (Exception e) {
        echo "Build failed due to: ${e}"
        currentBuild.result = 'FAILURE'
    } finally {
        echo 'Pipeline execution completed!'
    }
}
